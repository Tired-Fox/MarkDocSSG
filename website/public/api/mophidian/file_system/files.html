 75class Anchor:
 76    """Link representation of a header tag."""
 77
 78    def __init__(self, name: str, link: str, level: int) -> None:
 79        self._name = name
 80        self._level = level
 81        self._id = link.strip().lstrip("#")
 82
 83    @cached_property
 84    def link(self) -> str:
 85        """Link/href of the anchor."""
 86        return "#" + self._id
 87
 88    @property
 89    def name(self) -> str:
 90        """Name of the anchor."""
 91        return self._name
 92
 93    @property
 94    def level(self) -> int:
 95        """Anchor/header level."""
 96        return self._level
 97
 98    def __eq__(self, __o: object) -> bool:
 99        return bool(
100            isinstance(__o, self.__class__)
101            and self.name == __o.name
102            and self.link == __o.link
103            and self.level == __o.level
104        )
105
106    def __repr__(self) -> str:
107        return f"{self.name}({self.link!r}, {self.level})"

122        for anchor in self._children:

126        """Add an anchor object to the toc."""

204            file_info.groups() if file_info is not None else ("", None, None, "")

211        self.inherits = inherits is not None

215        self.build_dest()
216
217        # Page url

220        else:
221            self.relative_url = Path(self._dest).as_posix()
222
223        if self.relative_url != "./":
224            self.relative_url = "/" + self.relative_url.lstrip("/")
225        else:
226            self.relative_url = "/"

228    @cached_property

230        """Url of the page with the website root."""
231        url = Path(CONFIG.site.root).joinpath(self.relative_url.lstrip("/")).as_posix()
232
233        if url != ".":
234            if self.unique:
235                url = "/" + url
236            else:
237                url = "/" + url + "/"
238        else:
239            url = "/"
240        return url

242    def build_dest(self):
243        # Replace the file name with index.html
244        if self.file_name not in PAGE_IGNORE:
245            self._dest = (
246                Path("/".join(self._dest.split("/")[:-1]))
247                .joinpath("index.html")
248                .as_posix()
249            )
250        else:
251            self._dest = Path(self._dest).with_name(self.file_name + ".html").as_posix()

256        """Parsed phml ast of the file."""
257        raise Exception("Do not use base File class's ast property")
258

261        return Path(dest_dir).joinpath(self._dest.strip("/"))

263    def print(self, depth: int = 0) -> str:
264        """Colored terminal representation of the file."""
265        out = f"{' ' * depth}\x1b[34m{self.__class__.__name__}\x1b[0m > \
266\x1b[32m{self.file_name+self.extension!r}\x1b[0m"
267        if isinstance(self.children, list):
268            for child in list(self.children):
269                out += "\n" + child.print(depth + 4)
270        return out
271
272    def __repr__(self) -> str:
273        return f"{self.__class__.__name__}({self.path!r}, url={self.relative_url!r}, inherits={self.inherits!r}, \
274inherit_from={self.inherit_from!r})"

467    def parse_toc(self, toc: list):
468        """Parse the toc structure from the markdown parser and construct a toc object."""
469        result = TOC()
470
471        def flat_chilren(children: list):
472            for child in children:
473                result.append(Anchor(child["name"], child["id"], child["level"]))
474                if "children" in child and len(child["children"]) > 0:
475                    flat_chilren(child["children"])
476

478            result.append(Anchor(link["name"], link["id"], link["level"]))
479            if "children" in link and len(link["children"]) > 0:
480                flat_chilren(link["children"])
481

484    def _make_title(self) -> str:
485        with open(self.full_path, "r", encoding="utf-8") as mfile:
486            lines = mfile.readlines()
487            for i, line in enumerate(lines):
488                if line.strip() != "":
489                    header = match(r"(?P<hash>\s*# *.+)|(?P<block>=+)", line)
490                    if header is not None:
491                        header = header.groupdict()
492                        if header["hash"] is not None and header["hash"].strip() != "":
493                            return sub(r" *# *", "", header["hash"]).strip()
494                        if (
495                            header["block"] is not None
496                            and i > 0
497                            and lines[i - 1].strip() != ""
498                        ):
499                            return lines[i - 1].strip()
500
501        name = Path(self._dest).parent.as_posix().rsplit("/", 1)[-1]
502        if name.strip() in ["", "."]:
503            name = self.file_name
504
505        if name == "page":
506            name = "home"
507
508        return title(tokenize_name(name))

512            post = frontmatter.load(markdown_file)
