  7def merge(dest: dict, source: dict):
  8    """Merge a dictionary into another dictionary. Only adding keys that don't already exist. It
  9    will only replace the deepest keys.
 10
 11    Note:
 12        The method mutates the destination dict.
 13    """
 14    for key, value in source.items():
 15        if key in dest:
 16            if type(value) != type(dest[key]):
 17                dest[key] = value
 18            elif not isinstance(value, dict):
 19                dest[key] = value
 20            elif isinstance(value, dict):
 21                merge(dest[key], value)
 22        else:
 23            dest[key] = value
