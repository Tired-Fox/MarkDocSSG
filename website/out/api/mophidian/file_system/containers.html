 30def first(condition: Callable, collection: list | dict | tuple) -> Any:
 31    """Find the first match given the condition.
 32
 33    Return:
 34        Any | None: The found value or None if no value was found.
 35    """
 36    if isinstance(collection, dict):
 37        for value in collection.values():
 38            if condition(value):
 39                return value
 40    else:
 41        for value in collection:
 42            if condition(value):
 43                return value
 44    return None

 61    def remove(self, full_path: str):
 62        """Remove a specific file in file system given the files full path.
 63
 64        Args:
 65            full_path (str): The full path of the file to remove from the file system.
 66
 67        Raises:
 68            Exception: When no file is found that matches the full path.
 69        """
 70
 71        def iterate_children(container: Container) -> bool:
 72            for child in container.children:
 73                if isinstance(child, File) and child.full_path == full_path:
 74                    container.children.remove(child)
 75                    return True
 76                elif isinstance(child, Container):
 77                    result = iterate_children(child)
 78                    if result:
 79                        if len(child.children) == 0:
 80                            container.children.remove(child)
 81                        return result
 82            return False
 83
 84        if not iterate_children(self):
 85            raise Exception(f"No file found for full path {full_path!r}")

289    def build_nav(self) -> Nav:
290        nav = Nav("home")
291        nav_indexes = {"nav_pages": []}
292
293        pages = [page for page in self.renderable() if page.state != FileState.DELETED]

316                    unwrap_indecies(sub_nav, indecies[key])

321    def build_hierarchy(self):
322        """Build the relationships between layouts and pages."""
323
324        def iterate_layouts(current: Container, parent: Layout | None = None):
325            # Layouts without named inheritance
326            _layouts = [
327                layout
328                for layout in current.children
329                if isinstance(layout, Layout) and not layout.inherits
330            ]
331
332            # Layouts with named inheritance
333            _ilayouts = [
334                layout
335                for layout in current.children
336                if isinstance(layout, Layout) and layout.inherits
337            ]
338
339            # Directories and Groups
340            _containers = [
341                cont for cont in current.children if isinstance(cont, Container)
342            ]
343
344            if len(_layouts) > 1:
345                raise Exception(
346                    f"More than one layout for directory or group: {current.path}"
347                )
348
349            # If layout is in current directory assign as current parent layout
350            _layout = parent
351            if len(_layouts) == 1:
352                _layout = _layouts[0]
353                _layout.parent = parent
354
355            # Recursively process containers
356            for container in _containers:
357                iterate_layouts(container, _layout)
358
359            # For all named layouts find it's associated layout
360            for lyt in _ilayouts:
361                if lyt.inherits:
362                    lyt.parent = self.find_layout_by_name(lyt.inherit_from)
363
364        def iterate_pages(current: Container):
365            # Renderable files in the current directory
366            _pages = [page for page in current.children if isinstance(page, Renderable)]
367
368            # Directories and Groups
369            _containers = [
370                cont for cont in current.children if isinstance(cont, Container)
371            ]
372
373            for page in _pages:
374                if page.inherits:
375                    page.layout = self.find_layout_by_name(page.inherit_from)
376                else:
377                    page.layout = self.find_layout_by_path(page.parents)
378
379                # Link the page to all layouts that it uses
380                layout = page.layout
381                while layout is not None:
382                    layout.link_file(page)
383                    layout = layout.parent
384
385            # Recursively process containers
386            for container in _containers:
387                iterate_pages(container)
388
389        iterate_layouts(self)
390        iterate_pages(self)

392    def files(self, ext: str | list[str] | None = None) -> list[File]:
393        """List of all files in file system.

403
404        def build_compare():
405            if ext is None:
